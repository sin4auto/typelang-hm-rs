2// 役割: 抽象構文木(AST)の型定義と表示ユーティリティを管理する
3// 意図: パーサ・型推論・評価器が同じデータ構造を共有できるように整える

5//! AST モジュール
6//!
7//! 概要:
8//! - 構文解析で得た式やプログラム定義を列挙体・構造体で表現する。
9//! - 型推論器と評価器が追加の変換なしに読み取れる中立的な形を提供する。
10//! - 糖衣構文の解決や値の既定化は parser / repl 側に任せ、この層では正規化済みデータのみ扱う。
11
12use std::fmt;

14// 式を構成する列挙体
15#[derive(Clone, Debug, PartialEq)]
16/// 言語内の式を表す AST ノードの集合。
17pub enum Expr {
18    Var {
19        name: String,
20    },
21    IntLit {
22        value: i64,
23        base: IntBase,
24    },
25    FloatLit {
26        value: f64,
27    },
28    CharLit {
29        value: char,
30    },
31    StringLit {
32        value: String,
33    },
34    BoolLit {
35        value: bool,
36    },
37    ListLit {
38        items: Vec<Expr>,
39    },
40    TupleLit {
41        items: Vec<Expr>,
42    },
43    Lambda {
44        params: Vec<String>,
45        body: Box<Expr>,
46    },
47    LetIn {
48        bindings: Vec<(String, Vec<String>, Expr)>,
49        body: Box<Expr>,
50    },
51    If {
52        cond: Box<Expr>,
53        then_branch: Box<Expr>,
54        else_branch: Box<Expr>,
55    },
56    App {
57        func: Box<Expr>,
58        arg: Box<Expr>,
59    },
60    BinOp {
61        op: String,
62        left: Box<Expr>,
63        right: Box<Expr>,
64    },
65    Annot {
66        expr: Box<Expr>,
67        type_expr: TypeExpr,
68    },
69}
70
71#[derive(Clone, Copy, Debug, PartialEq, Eq)]
72/// 整数リテラルが使用した基数を保持する列挙体。
73pub enum IntBase {
74    Dec,
75    Hex,
76    Oct,
77    Bin,
78}

80// パーサが扱う型式ノード
81#[derive(Clone, Debug, PartialEq, Eq)]
82/// 構文上の型注釈を表すバリアント集合。
83pub enum TypeExpr {
84    TEVar(String),
85    TECon(String),
86    TEApp(Box<TypeExpr>, Box<TypeExpr>),
87    TEFun(Box<TypeExpr>, Box<TypeExpr>),
88    TEList(Box<TypeExpr>),
89    TETuple(Vec<TypeExpr>),
90}

92// 型クラス制約とシグマ型
93#[derive(Clone, Debug, PartialEq, Eq)]
94/// 型クラス名と対象の型変数を結び付ける制約。
95pub struct Constraint {
96    pub classname: String,
97    pub typevar: String,
98}

100#[derive(Clone, Debug, PartialEq, Eq)]
101/// 制約と型式を組み合わせたシグマ型。
102pub struct SigmaType {
103    pub constraints: Vec<Constraint>,
104    pub r#type: TypeExpr,
105}

108#[derive(Clone, Debug, PartialEq)]
109/// トップレベルで宣言された関数を保持するレコード。
110pub struct TopLevel {

112    pub params: Vec<String>,

123/// 式ノードを文字列表現へ整形する。
124impl fmt::Display for Expr {
125    /// デバッグしやすい括弧付きの表記に変換する。
126    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
127        match self {
128            Expr::Var { name } => write!(f, "{name}"),
129            Expr::IntLit { value, .. } => write!(f, "{value}"),
130            Expr::FloatLit { value } => write!(f, "{value}"),
131            Expr::CharLit { value } => write!(f, "'{value}'"),
132            Expr::StringLit { value } => write!(f, "\"{value}\""),
133            Expr::BoolLit { value } => write!(f, "{}", if *value { "True" } else { "False" }),
134            Expr::ListLit { items } => {
135                let parts: Vec<String> = items.iter().map(|e| format!("{}", e)).collect();
136                write!(f, "[{}]", parts.join(", "))
137            }
138            Expr::TupleLit { items } => {
139                let parts: Vec<String> = items.iter().map(|e| format!("{}", e)).collect();
140                write!(f, "({})", parts.join(", "))
141            }
142            Expr::Lambda { params, body } => write!(f, "\\{} -> {}", params.join(" "), body),
143            Expr::LetIn { bindings, body } => {
144                let mut parts = Vec::new();
145                for (n, ps, e) in bindings {
146                    if ps.is_empty() {
147                        parts.push(format!("{n} = {e}"));
148                    } else {
149                        parts.push(format!("{n} {} = {e}", ps.join(" ")));
150                    }
151                }
152                write!(f, "let {} in {}", parts.join("; "), body)
153            }
154            Expr::If {
155                cond,
156                then_branch,
157                else_branch,
158            } => {
159                write!(f, "if {} then {} else {}", cond, then_branch, else_branch)
160            }
161            Expr::App { func, arg } => write!(f, "({} {})", func, arg),
162            Expr::BinOp { op, left, right } => write!(f, "({} {} {})", left, op, right),
163            Expr::Annot { expr, type_expr } => write!(f, "({} :: {:?})", expr, type_expr),
164        }
165    }
166}
