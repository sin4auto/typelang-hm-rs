1//! 抽象構文木（AST）
2//!
3//! 目的:
4//! - 構文解析結果を評価/型推論で共用できる中立的な表現に落とし込む。
5//!
6//! 設計ノート:
7//! - 構文シュガーはここでは扱わず、parser/repl 側で正規化する。
8//! - 数値の基数（`IntBase`）は表示や一部の最適化のために保持する。
9//! - 型式（`TypeExpr`）は注釈・シグネチャ用で、型システムの `Type` とは分離。
10
11use std::fmt;

13// 式ノード
14#[derive(Clone, Debug, PartialEq)]
15pub enum Expr {
16    Var {
17        name: String,
18    },
19    IntLit {
20        value: i64,
21        base: IntBase,
22    },
23    FloatLit {
24        value: f64,
25    },
26    CharLit {
27        value: char,
28    },
29    StringLit {
30        value: String,
31    },
32    BoolLit {
33        value: bool,
34    },
35    ListLit {
36        items: Vec<Expr>,
37    },
38    TupleLit {
39        items: Vec<Expr>,
40    },
41    Lambda {
42        params: Vec<String>,
43        body: Box<Expr>,
44    },
45    LetIn {
46        bindings: Vec<(String, Vec<String>, Expr)>,
47        body: Box<Expr>,
48    },
49    If {
50        cond: Box<Expr>,
51        then_branch: Box<Expr>,
52        else_branch: Box<Expr>,
53    },
54    App {
55        func: Box<Expr>,
56        arg: Box<Expr>,
57    },
58    BinOp {
59        op: String,
60        left: Box<Expr>,
61        right: Box<Expr>,
62    },
63    Annot {
64        expr: Box<Expr>,
65        type_expr: TypeExpr,
66    },
67}
68
69#[derive(Clone, Copy, Debug, PartialEq, Eq)]
70pub enum IntBase {
71    Dec,
72    Hex,
73    Oct,
74    Bin,
75}

77// 型式（パーサ用）
78#[derive(Clone, Debug, PartialEq, Eq)]
79pub enum TypeExpr {
80    TEVar(String),
81    TECon(String),
82    TEApp(Box<TypeExpr>, Box<TypeExpr>),
83    TEFun(Box<TypeExpr>, Box<TypeExpr>),
84    TEList(Box<TypeExpr>),
85    TETuple(Vec<TypeExpr>),
86}

88// 制約と多相型（シグマ）

90pub struct Constraint {

115impl fmt::Display for Expr {
116    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
117        match self {
118            Expr::Var { name } => write!(f, "{name}"),
119            Expr::IntLit { value, .. } => write!(f, "{value}"),
120            Expr::FloatLit { value } => write!(f, "{value}"),
121            Expr::CharLit { value } => write!(f, "'{value}'"),
122            Expr::StringLit { value } => write!(f, "\"{value}\""),
123            Expr::BoolLit { value } => write!(f, "{}", if *value { "True" } else { "False" }),
124            Expr::ListLit { items } => {
125                let parts: Vec<String> = items.iter().map(|e| format!("{}", e)).collect();
126                write!(f, "[{}]", parts.join(", "))
127            }
128            Expr::TupleLit { items } => {
129                let parts: Vec<String> = items.iter().map(|e| format!("{}", e)).collect();
130                write!(f, "({})", parts.join(", "))
131            }
132            Expr::Lambda { params, body } => write!(f, "\\{} -> {}", params.join(" "), body),
133            Expr::LetIn { bindings, body } => {
134                let mut parts = Vec::new();
135                for (n, ps, e) in bindings {
136                    if ps.is_empty() {
137                        parts.push(format!("{n} = {e}"));
138                    } else {
139                        parts.push(format!("{n} {} = {e}", ps.join(" ")));
140                    }
141                }
142                write!(f, "let {} in {}", parts.join("; "), body)
143            }
144            Expr::If {
145                cond,
146                then_branch,
147                else_branch,
148            } => {
149                write!(f, "if {} then {} else {}", cond, then_branch, else_branch)
150            }
151            Expr::App { func, arg } => write!(f, "({} {})", func, arg),
152            Expr::BinOp { op, left, right } => write!(f, "({} {} {})", left, op, right),
153            Expr::Annot { expr, type_expr } => write!(f, "({} :: {:?})", expr, type_expr),
154        }
155    }
156}
