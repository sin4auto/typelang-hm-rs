5//! 構文解析モジュール
6//!
7//! - EBNF で定義された文法に従ってプログラム・式・型注釈を解析する。
8//! - 演算子の結合規則・優先順位は `cmp > add > mul > pow > app` の順でハンドコードする。
9//! - 単項マイナスなどの糖衣は `0 - x` など正規化した AST へ変換する。

17/// 再帰下降パーサの進行状態を保持する構造体。
18pub struct Parser {
19    ts: Vec<Token>,
20    i: usize,
21}

23#[derive(Clone, Copy)]
24enum Assoc {
25    Left,
26    Right,
27    Non,
28}
29
30struct InfixSpec {
31    tokens: &'static [TokenKind],
32    assoc: Assoc,
33}
34
35impl InfixSpec {
36    fn contains(&self, kind: &TokenKind) -> bool {
37        self.tokens.iter().any(|tk| tk == kind)
38    }
39}

41const INFIX_LEVELS: &[InfixSpec] = &[
42    InfixSpec {
43        tokens: &[
44            TokenKind::EQ,
45            TokenKind::NE,
46            TokenKind::LT,
47            TokenKind::LE,
48            TokenKind::GT,
49            TokenKind::GE,
50        ],
51        assoc: Assoc::Non,
52    },
53    InfixSpec {
54        tokens: &[TokenKind::PLUS, TokenKind::MINUS],
55        assoc: Assoc::Left,
56    },
57    InfixSpec {
58        tokens: &[TokenKind::STAR, TokenKind::SLASH],
59        assoc: Assoc::Left,
60    },
61    InfixSpec {
62        tokens: &[TokenKind::CARET, TokenKind::DBLSTAR],
63        assoc: Assoc::Right,
64    },
65];

67/// `Parser` が提供する各種解析メソッド。
68impl Parser {
69    /// トークン列から新しいパーサインスタンスを構築する。
70    pub fn new(tokens: Vec<Token>) -> Self {
71        Self { ts: tokens, i: 0 }
72    }
73    /// 現在位置のトークンを参照する。
74    fn peek(&self) -> &Token {
75        &self.ts[self.i]
76    }
77    /// 現在位置のトークンを消費して返す。
78    fn pop_any(&mut self) -> Token {
79        let t = self.ts[self.i].clone();
80        self.i += 1;
81        t
82    }
83    /// 指定した種別のトークンを期待しつつ消費する。
84    fn pop(&mut self, kind: TokenKind) -> Result<Token, ParseError> {
85        let t = self.peek().clone();
86        if t.kind != kind {
87            return Err(ParseError::at(
88                "PAR001",
89                format!("{:?} を期待しましたが {:?} ({})", kind, t.kind, t.value),
90                Some(t.pos),
91                Some(t.line),
92                Some(t.col),
93            ));
94        }
95        self.i += 1;
96        Ok(t)
97    }
98    /// 指定した種別が先頭にあれば消費し、なければ `None` を返す。
99    fn accept(&mut self, kind: TokenKind) -> Option<Token> {
100        if self.peek().kind == kind {
101            let t = self.pop_any();
102            Some(t)
103        } else {
104            None
105        }
106    }

108    /// プログラム全体（`program := { toplevel }`）を解析する。
109    ///
110    /// # Examples
111    /// ```
112    /// use typelang::parser::{self, Parser};
113    /// let src = "square :: Num a => a -> a;\nlet square x = x * x;";
114    /// let tokens = typelang::lexer::lex(src).unwrap();
115    /// let mut p = Parser::new(tokens);
116    /// let prog = p.parse_program().unwrap();
117    /// assert_eq!(prog.decls.len(), 1);
118    /// ```
119    pub fn parse_program(&mut self) -> Result<Program, ParseError> {
120        let mut decls = Vec::new();
121        while self.peek().kind != TokenKind::EOF {
122            let mut sig: Option<SigmaType> = None;
123            // 先読み: varid '::' ...
124            let save = self.i;
125            if self.peek().kind == TokenKind::VARID {
126                let _name_tok = self.pop_any();
127                if self.accept(TokenKind::DCOLON).is_some() {
128                    sig = Some(self.parse_sigma_type()?);
129                    self._expect_semicolon_optional()?;
130                } else {
131                    self.i = save;
132                }
133            }
134            // let binding
135            self.pop(TokenKind::LET)?;
136            let name_tok = self.pop(TokenKind::VARID)?;
137            let mut params: Vec<String> = Vec::new();
138            while self.peek().kind == TokenKind::VARID {
139                params.push(self.pop_any().value);
140            }
141            self.pop(TokenKind::EQUAL)?;
142            let expr = self.parse_expr()?;
143            self._expect_semicolon_optional()?;
144            decls.push(TopLevel {
145                name: name_tok.value,
146                params,
147                expr,
148                signature: sig,
149            });
150        }
151        Ok(Program { decls })
152    }

154    /// 文末のセミコロンがあれば消費する（省略可）。
155    fn _expect_semicolon_optional(&mut self) -> Result<(), ParseError> {
156        self.accept(TokenKind::SEMI);
157        Ok(())
158    }

160    /// 型注釈を含む単一式（`expr := core_expr ['::' type]`）を解析する。
161    ///
162    /// # Examples
163    /// ```
164    /// use typelang::parser;
165    /// let e = parser::parse_expr("1 + 2").unwrap();
166    /// match e { typelang::ast::Expr::BinOp{..} => (), _ => unreachable!() }
167    /// ```
168    pub fn parse_expr(&mut self) -> Result<Expr, ParseError> {
169        let expr = self.parse_expr_no_annot()?;
170        if self.accept(TokenKind::DCOLON).is_some() {
171            let ty = self.parse_type()?;
172            Ok(Expr::Annot {
173                expr: Box::new(expr),
174                type_expr: ty,
175            })
176        } else {
177            Ok(expr)
178        }
179    }

181    /// 型注釈を除いた式（ラムダ/let-in/if/二項式）を解析する。
182    fn parse_expr_no_annot(&mut self) -> Result<Expr, ParseError> {
183        match self.peek().kind.clone() {
184            TokenKind::LAMBDA => self.parse_lambda(),
185            TokenKind::LET => self.parse_let_in(),
186            TokenKind::IF => self.parse_if(),
187            _ => self.parse_infix_level(0),
188        }
189    }

191    /// ラムダ式（`\ params -> expr`）を解析する。
192    fn parse_lambda(&mut self) -> Result<Expr, ParseError> {
193        self.pop(TokenKind::LAMBDA)?;
194        let mut params: Vec<String> = Vec::new();
195        while self.peek().kind == TokenKind::VARID {
196            params.push(self.pop_any().value);
197        }
198        self.pop(TokenKind::ARROW)?;
199        let body = self.parse_expr()?;
200        Ok(Expr::Lambda {
201            params,
202            body: Box::new(body),
203        })
204    }
205
206    /// `let ... in ...` 式を解析し、束縛を収集する。
207    fn parse_let_in(&mut self) -> Result<Expr, ParseError> {
208        self.pop(TokenKind::LET)?;
209        let mut bindings: Vec<(String, Vec<String>, Expr)> = Vec::new();
210        loop {
211            let name = self.pop(TokenKind::VARID)?.value;
212            let mut params: Vec<String> = Vec::new();
213            while self.peek().kind == TokenKind::VARID {
214                params.push(self.pop_any().value);
215            }
216            self.pop(TokenKind::EQUAL)?;
217            let e = self.parse_expr()?;
218            bindings.push((name, params, e));
219            if self.accept(TokenKind::SEMI).is_none() {
220                break;
221            }
222        }
223        self.pop(TokenKind::IN)?;
224        let body = self.parse_expr()?;
225        Ok(Expr::LetIn {
226            bindings,
227            body: Box::new(body),
228        })
229    }
230
231    /// `if ... then ... else ...` 式を解析する。
232    fn parse_if(&mut self) -> Result<Expr, ParseError> {
233        self.pop(TokenKind::IF)?;
234        let cond = self.parse_expr()?;
235        self.pop(TokenKind::THEN)?;
236        let th = self.parse_expr()?;
237        self.pop(TokenKind::ELSE)?;
238        let el = self.parse_expr()?;
239        Ok(Expr::If {
240            cond: Box::new(cond),
241            then_branch: Box::new(th),
242            else_branch: Box::new(el),
243        })
244    }

246    fn parse_infix_level(&mut self, level: usize) -> Result<Expr, ParseError> {
247        if level >= INFIX_LEVELS.len() {
248            return self.parse_app();
249        }
250        let spec = &INFIX_LEVELS[level];
251        match spec.assoc {
252            Assoc::Left => self.parse_infix_left(level, spec),
253            Assoc::Right => self.parse_infix_right(level, spec),
254            Assoc::Non => self.parse_infix_non(level, spec),
255        }
256    }
257
258    fn parse_infix_left(&mut self, level: usize, spec: &InfixSpec) -> Result<Expr, ParseError> {
259        let mut left = self.parse_infix_level(level + 1)?;
260        while spec.contains(&self.peek().kind) {
261            let op = self.pop_any().value;
262            let right = self.parse_infix_level(level + 1)?;
263            left = Expr::BinOp {
264                op,
265                left: Box::new(left),
266                right: Box::new(right),
267            };
268        }
269        Ok(left)
270    }
271
272    fn parse_infix_right(&mut self, level: usize, spec: &InfixSpec) -> Result<Expr, ParseError> {
273        let left = self.parse_infix_level(level + 1)?;
274        if spec.contains(&self.peek().kind) {
275            let op = self.pop_any().value;
276            let right = self.parse_infix_level(level)?;
277            Ok(Expr::BinOp {
278                op,
279                left: Box::new(left),
280                right: Box::new(right),
281            })
282        } else {
283            Ok(left)
284        }
285    }
286
287    fn parse_infix_non(&mut self, level: usize, spec: &InfixSpec) -> Result<Expr, ParseError> {
288        let left = self.parse_infix_level(level + 1)?;
289        if spec.contains(&self.peek().kind) {
290            let op = self.pop_any().value;
291            let right = self.parse_infix_level(level + 1)?;
292            Ok(Expr::BinOp {
293                op,
294                left: Box::new(left),
295                right: Box::new(right),
296            })
297        } else {
298            Ok(left)
299        }
300    }

302    /// 関数適用（左結合）を解析する。
303    fn parse_app(&mut self) -> Result<Expr, ParseError> {
304        let mut left = self.parse_atom()?;
305        while let TokenKind::INT
306        | TokenKind::HEX
307        | TokenKind::OCT
308        | TokenKind::BIN
309        | TokenKind::FLOAT
310        | TokenKind::CHAR
311        | TokenKind::STRING
312        | TokenKind::VARID
313        | TokenKind::UNDERSCORE
314        | TokenKind::QMARK
315        | TokenKind::LPAREN
316        | TokenKind::LBRACK
317        | TokenKind::TRUE
318        | TokenKind::FALSE = self.peek().kind
319        {
320            let right = self.parse_atom()?;
321            left = Expr::App {
322                func: Box::new(left),
323                arg: Box::new(right),
324            };
325        }
326        Ok(left)
327    }

329    /// リテラル・識別子・括弧を含む原子式を解析する。
330    fn parse_atom(&mut self) -> Result<Expr, ParseError> {
331        let t = self.peek().clone();
332        match t.kind {
333            TokenKind::MINUS => {
334                self.pop_any();
335                let rhs = self.parse_atom()?;
336                Ok(Expr::BinOp {
337                    op: "-".into(),
338                    left: Box::new(Expr::IntLit {
339                        value: 0,
340                        base: IntBase::Dec,
341                    }),
342                    right: Box::new(rhs),
343                })
344            }
345            TokenKind::INT | TokenKind::HEX | TokenKind::OCT | TokenKind::BIN => {
346                self.pop_any();
347                let s = t.value.clone();
348                let parsed: Result<i64, _> = match t.kind {
349                    TokenKind::HEX => i64::from_str_radix(&s[2..], 16),
350                    TokenKind::OCT => i64::from_str_radix(&s[2..], 8),
351                    TokenKind::BIN => i64::from_str_radix(&s[2..], 2),
352                    _ => s.parse::<i64>(),
353                };
354                match parsed {
355                    Ok(v) => {
356                        let base = match t.kind {
357                            TokenKind::HEX => IntBase::Hex,
358                            TokenKind::OCT => IntBase::Oct,
359                            TokenKind::BIN => IntBase::Bin,
360                            _ => IntBase::Dec,
361                        };
362                        Ok(Expr::IntLit { value: v, base })
363                    }
364                    Err(_) => Err(ParseError::new(
365                        "PAR210",
366                        "整数リテラルが範囲外",
367                        Some(t.pos),
368                    )),
369                }
370            }
371            TokenKind::FLOAT => {
372                self.pop_any();
373                match t.value.parse::<f64>() {
374                    Ok(v) => Ok(Expr::FloatLit { value: v }),
375                    Err(_) => Err(ParseError::at(
376                        "PAR220",
377                        "浮動小数リテラルが不正",
378                        Some(t.pos),
379                        Some(t.line),
380                        Some(t.col),
381                    )),
382                }
383            }
384            TokenKind::CHAR => {
385                self.pop_any();
386                let ch = decode_char(&t.value)?;
387                Ok(Expr::CharLit { value: ch })
388            }
389            TokenKind::STRING => {
390                self.pop_any();
391                let s = decode_string(&t.value)?;
392                Ok(Expr::StringLit { value: s })
393            }
394            TokenKind::TRUE => {
395                self.pop_any();
396                Ok(Expr::BoolLit { value: true })
397            }
398            TokenKind::FALSE => {
399                self.pop_any();
400                Ok(Expr::BoolLit { value: false })
401            }
402            TokenKind::VARID => {
403                self.pop_any();
404                Ok(Expr::Var { name: t.value })
405            }
406            TokenKind::QMARK => {
407                self.pop_any();
408                let name = self.pop(TokenKind::VARID)?.value;
409                Ok(Expr::Var {
410                    name: format!("?{}", name),
411                })
412            }
413            TokenKind::UNDERSCORE => {
414                self.pop_any();
415                Ok(Expr::Var { name: "_".into() })
416            }
417            TokenKind::LPAREN => {
418                self.pop_any();
419                let e = self.parse_expr()?;
420                if self.accept(TokenKind::COMMA).is_some() {
421                    let mut items = vec![e, self.parse_expr()?];
422                    while self.accept(TokenKind::COMMA).is_some() {
423                        items.push(self.parse_expr()?);
424                    }
425                    self.pop(TokenKind::RPAREN)?;
426                    Ok(Expr::TupleLit { items })
427                } else {
428                    self.pop(TokenKind::RPAREN)?;
429                    Ok(e)
430                }
431            }
432            TokenKind::LBRACK => {
433                self.pop_any();
434                let mut items = Vec::new();
435                if self.peek().kind != TokenKind::RBRACK {
436                    items.push(self.parse_expr()?);
437                    while self.accept(TokenKind::COMMA).is_some() {
438                        items.push(self.parse_expr()?);
439                    }
440                }
441                self.pop(TokenKind::RBRACK)?;
442                Ok(Expr::ListLit { items })
443            }
444            _ => Err(ParseError::new(
445                "PAR002",
446                format!("不正なトークン: {:?} {}", t.kind, t.value),
447                Some(t.pos),
448            )),
449        }
450    }

452    // ===== 型式 =====
453    /// 制約付きシグマ型（`context => type`）を解析する。
454    pub fn parse_sigma_type(&mut self) -> Result<SigmaType, ParseError> {
455        // 先読み context '=>'
456        let mut constraints: Vec<AConstraint> = Vec::new();
457        if matches!(self.peek().kind, TokenKind::CONID | TokenKind::LPAREN) {
458            let save = self.i;
459            if let Some(cs) = self.try_parse_context()? {
460                constraints = cs;
461                self.pop(TokenKind::DARROW)?;
462            } else {
463                self.i = save;
464            }
465        }
466        let ty = self.parse_type()?;
467        Ok(SigmaType {
468            constraints,
469            r#type: ty,
470        })
471    }
472
473    /// 制約コンテキストを先読みし、存在すれば解析して返す。
474    fn try_parse_context(&mut self) -> Result<Option<Vec<AConstraint>>, ParseError> {
475        match self.peek().kind {
476            TokenKind::CONID => {
477                let c = self.parse_constraint()?;
478                if self.peek().kind == TokenKind::DARROW {
479                    Ok(Some(vec![c]))
480                } else {
481                    Ok(None)
482                }
483            }
484            TokenKind::LPAREN => {
485                let save = self.i;
486                self.pop(TokenKind::LPAREN)?;
487                let mut cs = vec![self.parse_constraint()?];
488                while self.accept(TokenKind::COMMA).is_some() {
489                    cs.push(self.parse_constraint()?);
490                }
491                if self.accept(TokenKind::RPAREN).is_none() {
492                    self.i = save;
493                    return Ok(None);
494                }
495                if self.peek().kind == TokenKind::DARROW {
496                    Ok(Some(cs))
497                } else {
498                    self.i = save;
499                    Ok(None)
500                }
501            }
502            _ => Ok(None),
503        }
504    }

506    /// `Class var` 形式の型クラス制約を解析する。
507    fn parse_constraint(&mut self) -> Result<AConstraint, ParseError> {
508        let cls = self.pop(TokenKind::CONID)?.value;
509        let tv = self.pop(TokenKind::VARID)?.value;
510        Ok(AConstraint {
511            classname: cls,
512            typevar: tv,
513        })
514    }

516    /// 関数矢印を含む型式を解析する。
517    pub fn parse_type(&mut self) -> Result<TypeExpr, ParseError> {
518        let left = self.parse_type_app()?;
519        if self.accept(TokenKind::ARROW).is_some() {
520            let right = self.parse_type()?;
521            Ok(TypeExpr::TEFun(Box::new(left), Box::new(right)))
522        } else {
523            Ok(left)
524        }
525    }
526
527    /// 型適用（左結合）を解析する。
528    fn parse_type_app(&mut self) -> Result<TypeExpr, ParseError> {
529        let mut left = self.parse_type_atom()?;
530        while let TokenKind::CONID | TokenKind::VARID | TokenKind::LPAREN | TokenKind::LBRACK =
531            self.peek().kind
532        {
533            let right = self.parse_type_atom()?;
534            left = TypeExpr::TEApp(Box::new(left), Box::new(right));
535        }
536        Ok(left)
537    }
538
539    /// 型式の原子要素（変数・コンストラクタ・タプル・リスト）を解析する。
540    fn parse_type_atom(&mut self) -> Result<TypeExpr, ParseError> {
541        let t = self.peek().clone();
542        match t.kind {
543            TokenKind::VARID => {
544                self.pop_any();
545                Ok(TypeExpr::TEVar(t.value))
546            }
547            TokenKind::CONID => {
548                self.pop_any();
549                Ok(TypeExpr::TECon(t.value))
550            }
551            TokenKind::LPAREN => {
552                self.pop_any();
553                let ty = self.parse_type()?;
554                if self.accept(TokenKind::COMMA).is_some() {
555                    let mut items = vec![ty, self.parse_type()?];
556                    while self.accept(TokenKind::COMMA).is_some() {
557                        items.push(self.parse_type()?);
558                    }
559                    self.pop(TokenKind::RPAREN)?;
560                    Ok(TypeExpr::TETuple(items))
561                } else {
562                    self.pop(TokenKind::RPAREN)?;
563                    Ok(ty)
564                }
565            }
566            TokenKind::LBRACK => {
567                self.pop_any();
568                let inner = self.parse_type()?;
569                self.pop(TokenKind::RBRACK)?;
570                Ok(TypeExpr::TEList(Box::new(inner)))
571            }
572            _ => Err(ParseError::at(
573                "PAR102",
574                format!("型の原子が不正: {:?} {}", t.kind, t.value),
575                Some(t.pos),
576                Some(t.line),
577                Some(t.col),
578            )),
579        }
580    }
581}

584/// ソース上の文字列リテラルをアンエスケープし、内容だけを返す。
585fn decode_string(quoted: &str) -> Result<String, ParseError> {
586    // quoted には "..." を含む
587    if !quoted.starts_with('"') || !quoted.ends_with('"') {
588        return Err(ParseError::new("PAR201", "文字列リテラルが不正", None));
589    }
590    let s = &quoted[1..quoted.len() - 1];
591    let mut out = String::new();
592    let mut chars = s.chars();
593    while let Some(ch) = chars.next() {
594        if ch == '\\' {
595            // エスケープシーケンスを復号する
596            let Some(e) = chars.next() else {
597                return Err(ParseError::new("PAR202", "末尾のバックスラッシュ", None));
598            };
599            match e {
600                'n' => out.push('\n'),
601                'r' => out.push('\r'),
602                't' => out.push('\t'),
603                '\\' => out.push('\\'),
604                '\'' => out.push('\''),
605                '"' => out.push('"'),
606                _ => out.push(e),
607            }
608        } else {
609            out.push(ch);
610        }
611    }
612    Ok(out)
613}
614
615/// ソース上の文字リテラルをアンエスケープして単一文字を返す。
616fn decode_char(quoted: &str) -> Result<char, ParseError> {
617    if !quoted.starts_with('\'') || !quoted.ends_with('\'') {
618        return Err(ParseError::new("PAR204", "文字リテラルが不正", None));
619    }
620    let s = &quoted[1..quoted.len() - 1];
621    let ch = if s.starts_with('\\') {
622        match s.chars().nth(1) {
623            Some('n') => '\n',
624            Some('r') => '\r',
625            Some('t') => '\t',
626            Some('\\') => '\\',
627            Some('\'') => '\'',
628            Some('"') => '"',
629            Some(x) => x,
630            None => return Err(ParseError::new("PAR203", "空のエスケープ", None)),
631        }
632    } else {
633        s.chars()
634            .next()
635            .ok_or_else(|| ParseError::new("PAR205", "空の文字", None))?
636    };
637    Ok(ch)
638}

641/// ソース文字列を完全に解析し、`Program` を返す高水準 API。
642pub fn parse_program(src: &str) -> Result<Program, ParseError> {
643    let ts = lex(src).map_err(|e| ParseError::new("PAR100", format!("lex error: {}", e), None))?;
644    Parser::new(ts).parse_program()
645}
646/// ソース文字列から単一の式を解析する高水準 API。
647pub fn parse_expr(src: &str) -> Result<Expr, ParseError> {
648    let ts = lex(src).map_err(|e| ParseError::new("PAR100", format!("lex error: {}", e), None))?;
649    let mut p = Parser::new(ts);
650    let e = p.parse_expr()?;
651    if p.peek().kind != TokenKind::EOF {
652        let t = p.peek().clone();
653        return Err(ParseError::at(
654            "PAR090",
655            "余分なトークンが残っています",
656            Some(t.pos),
657            Some(t.line),
658            Some(t.col),
659        ));
660    }
661    Ok(e)
662}
