18/// 式を推論し、必要に応じて defaulting を適用した `QualType` を返す。
19pub(crate) fn infer_qual_type(
20    type_env: &TypeEnv,
21    class_env: &ClassEnv,
22    expr: &A::Expr,
23    defaulting_on: bool,
24) -> Result<QualType, TypeError> {
25    let mut st = InferState {
26        supply: Default::default(),
27        subst: Default::default(),
28    };
29    let (subst, qual) = infer_expr(type_env, class_env, &mut st, expr)?;
30    let mut applied = qual.apply_subst(&subst);
31    if defaulting_on {
32        applied = apply_defaulting_simple(&applied);
33    }
34    Ok(applied)
35}

37/// 評価結果から復旧用の型スキームを構築する。
38pub(crate) fn fallback_scheme_from_value(type_env: &TypeEnv, value: &Value) -> Scheme {
39    let fallback_type = fallback_type_from_value(value);
40    generalize(type_env, qualify(fallback_type, vec![]))
41}

43/// 評価結果に対応する `QualType` を返す。
44pub(crate) fn fallback_qual_from_value(value: &Value) -> QualType {
45    QualType {
46        constraints: vec![],
47        r#type: fallback_type_from_value(value),
48    }
49}

51/// 推論失敗時などに利用するフォールバック型を算出する。
52fn fallback_type_from_value(value: &Value) -> Type {
53    match value {
54        Value::Int(_) => Type::TCon(TCon { name: "Int".into() }),
55        Value::Double(_) => Type::TCon(TCon {
56            name: "Double".into(),
57        }),
58        Value::Bool(_) => Type::TCon(TCon {
59            name: "Bool".into(),
60        }),
61        Value::Char(_) => Type::TCon(TCon {
62            name: "Char".into(),
63        }),
64        Value::String(_) => t_string(),
65        Value::List(_) | Value::Tuple(_) => Type::TTuple(TTuple { items: vec![] }),
66        Value::Closure { .. } | Value::Prim1(_) | Value::Prim2(_) => {
67            Type::TTuple(TTuple { items: vec![] })
68        }
69    }
70}
