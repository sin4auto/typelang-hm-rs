5/// (^) で右辺が負の整数のときに (** n.0) へ正規化して
6/// 型推論を安定化させる簡易変換。
7pub(crate) fn normalize_expr(e: &A::Expr) -> A::Expr {
8    use A::Expr::*;
9    fn neg_int(e: &A::Expr) -> Option<i64> {
10        if let A::Expr::BinOp { op, left, right } = e {
11            if op == "-" {
12                if let A::Expr::IntLit { value: 0, .. } = **left {
13                    if let A::Expr::IntLit { value: n, .. } = **right {
14                        return Some(-n);
15                    }
16                }
17            }
18        }
19        None
20    }
21    match e {
22        BinOp { op, left, right } => {
23            let l = Box::new(normalize_expr(left));
24            let r = Box::new(normalize_expr(right));
25            if op == "^" {
26                if let Some(n) = neg_int(&r) {
27                    return BinOp {
28                        op: "**".into(),
29                        left: l,
30                        right: Box::new(FloatLit { value: n as f64 }),
31                    };
32                }
33            }
34            BinOp {
35                op: op.clone(),
36                left: l,
37                right: r,
38            }
39        }
40        App { func, arg } => App {
41            func: Box::new(normalize_expr(func)),
42            arg: Box::new(normalize_expr(arg)),
43        },
44        Lambda { params, body } => Lambda {
45            params: params.clone(),
46            body: Box::new(normalize_expr(body)),
47        },
48        LetIn { bindings, body } => {
49            let bs: Vec<_> = bindings
50                .iter()
51                .map(|(n, ps, ex)| (n.clone(), ps.clone(), normalize_expr(ex)))
52                .collect();
53            LetIn {
54                bindings: bs,
55                body: Box::new(normalize_expr(body)),
56            }
57        }
58        If {
59            cond,
60            then_branch,
61            else_branch,
62        } => If {
63            cond: Box::new(normalize_expr(cond)),
64            then_branch: Box::new(normalize_expr(then_branch)),
65            else_branch: Box::new(normalize_expr(else_branch)),
66        },
67        Annot { expr, type_expr } => Annot {
68            expr: Box::new(normalize_expr(expr)),
69            type_expr: type_expr.clone(),
70        },
71        ListLit { items } => ListLit {
72            items: items.iter().map(normalize_expr).collect(),
73        },
74        TupleLit { items } => TupleLit {
75            items: items.iter().map(normalize_expr).collect(),
76        },
77        other => other.clone(),
78    }
79}

81#[cfg(test)]
82mod tests {
83    use super::normalize_expr;
84    use crate::ast as A;

86    #[test]
87    fn normalize_pow_with_negative_int_exponent_changes_to_starstar() {
88        let e = A::Expr::BinOp {
89            op: "^".into(),
90            left: Box::new(A::Expr::IntLit {
91                value: 2,
92                base: A::IntBase::Dec,
93            }),
94            right: Box::new(A::Expr::BinOp {
95                op: "-".into(),
96                left: Box::new(A::Expr::IntLit {
97                    value: 0,
98                    base: A::IntBase::Dec,
99                }),
100                right: Box::new(A::Expr::IntLit {
101                    value: 3,
102                    base: A::IntBase::Dec,
103                }),
104            }),
105        };
106        let n = normalize_expr(&e);
107        match n {
108            A::Expr::BinOp { op, left: _, right } => {
109                assert_eq!(op, "**");
110                assert!(
111                    matches!(*right, A::Expr::FloatLit { value } if (value - (-3.0)).abs() < 1e-12)
112                );
113            }
114            _ => panic!("not normalized to BinOp"),
115        }
116    }
117
118    #[test]
119    fn normalize_keeps_other_ops_untouched() {
120        let e = A::Expr::BinOp {
121            op: "+".into(),
122            left: Box::new(A::Expr::IntLit {
123                value: 1,
124                base: A::IntBase::Dec,
125            }),
126            right: Box::new(A::Expr::IntLit {
127                value: 2,
128                base: A::IntBase::Dec,
129            }),
130        };
131        let n = normalize_expr(&e);
132        match n {
133            A::Expr::BinOp { op, .. } => assert_eq!(op, "+"),
134            _ => panic!("unexpected"),
135        }
136    }

138    #[test]
139    fn normalize_recurse_into_collections() {
140        // [(2 ^ -1)] のような入れ子でも再帰的に正規化されること
141        let e = A::Expr::ListLit {
142            items: vec![A::Expr::BinOp {
143                op: "^".into(),
144                left: Box::new(A::Expr::IntLit {
145                    value: 2,
146                    base: A::IntBase::Dec,
147                }),
148                right: Box::new(A::Expr::BinOp {
149                    op: "-".into(),
150                    left: Box::new(A::Expr::IntLit {
151                        value: 0,
152                        base: A::IntBase::Dec,
153                    }),
154                    right: Box::new(A::Expr::IntLit {
155                        value: 1,
156                        base: A::IntBase::Dec,
157                    }),
158                }),
159            }],
160        };
161        let n = normalize_expr(&e);
162        if let A::Expr::ListLit { items } = n {
163            if let A::Expr::BinOp { op, .. } = &items[0] {
164                assert_eq!(op, "**");
165                return;
166            }
167        }
168        panic!("not normalized in collection");
169    }
