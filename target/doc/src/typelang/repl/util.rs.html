5//! REPL 内部で共有する式変換ユーティリティを提供するモジュール。
6//! 推論が安定するようユーザー入力を事前整形する。
7
8use crate::ast as A;
9/// ユーザー入力の式を正規化し、負のべき乗などを推論しやすい形に直す。
10pub(crate) fn normalize_expr(e: &A::Expr) -> A::Expr {
11    use A::Expr::*;
12    /// `0 - n` 形式の負整数リテラルを検出する補助関数。
13    fn neg_int(e: &A::Expr) -> Option<i64> {
14        if let A::Expr::BinOp { op, left, right } = e {
15            if op == "-" {
16                if let A::Expr::IntLit { value: 0, .. } = **left {
17                    if let A::Expr::IntLit { value: n, .. } = **right {
18                        return Some(-n);
19                    }
20                }
21            }
22        }
23        None
24    }
25    match e {
26        BinOp { op, left, right } => {
27            let l = Box::new(normalize_expr(left));
28            let r = Box::new(normalize_expr(right));
29            if op == "^" {
30                if let Some(n) = neg_int(&r) {
31                    return BinOp {
32                        op: "**".into(),
33                        left: l,
34                        right: Box::new(FloatLit { value: n as f64 }),
35                    };
36                }
37            }
38            BinOp {
39                op: op.clone(),
40                left: l,
41                right: r,
42            }
43        }
44        App { func, arg } => App {
45            func: Box::new(normalize_expr(func)),
46            arg: Box::new(normalize_expr(arg)),
47        },
48        Lambda { params, body } => Lambda {
49            params: params.clone(),
50            body: Box::new(normalize_expr(body)),
51        },
52        LetIn { bindings, body } => {
53            let bs: Vec<_> = bindings
54                .iter()
55                .map(|(n, ps, ex)| (n.clone(), ps.clone(), normalize_expr(ex)))
56                .collect();
57            LetIn {
58                bindings: bs,
59                body: Box::new(normalize_expr(body)),
60            }
61        }
62        If {
63            cond,
64            then_branch,
65            else_branch,
66        } => If {
67            cond: Box::new(normalize_expr(cond)),
68            then_branch: Box::new(normalize_expr(then_branch)),
69            else_branch: Box::new(normalize_expr(else_branch)),
70        },
71        Annot { expr, type_expr } => Annot {
72            expr: Box::new(normalize_expr(expr)),
73            type_expr: type_expr.clone(),
74        },
75        ListLit { items } => ListLit {
76            items: items.iter().map(normalize_expr).collect(),
77        },
78        TupleLit { items } => TupleLit {
79            items: items.iter().map(normalize_expr).collect(),
80        },
81        other => other.clone(),
82    }
83}

85#[cfg(test)]
86mod tests {
87    use super::normalize_expr;
88    use crate::ast as A;
89
90    #[test]
91    /// 負の整数指数が `^` から `**` へ変換されることを確認する。
92    fn normalize_pow_with_negative_int_exponent_changes_to_starstar() {
93        let e = A::Expr::BinOp {
94            op: "^".into(),
95            left: Box::new(A::Expr::IntLit {
96                value: 2,
97                base: A::IntBase::Dec,
98            }),
99            right: Box::new(A::Expr::BinOp {
100                op: "-".into(),
101                left: Box::new(A::Expr::IntLit {
102                    value: 0,
103                    base: A::IntBase::Dec,
104                }),
105                right: Box::new(A::Expr::IntLit {
106                    value: 3,
107                    base: A::IntBase::Dec,
108                }),
109            }),
110        };
111        let n = normalize_expr(&e);
112        match n {
113            A::Expr::BinOp { op, left: _, right } => {
114                assert_eq!(op, "**");
115                assert!(
116                    matches!(*right, A::Expr::FloatLit { value } if (value - (-3.0)).abs() < 1e-12)
117                );
118            }
119            _ => panic!("not normalized to BinOp"),
120        }
121    }
122
123    #[test]
124    /// 対象外の演算子がそのまま残ることを検証する。
125    fn normalize_keeps_other_ops_untouched() {
126        let e = A::Expr::BinOp {
127            op: "+".into(),
128            left: Box::new(A::Expr::IntLit {
129                value: 1,
130                base: A::IntBase::Dec,
131            }),
132            right: Box::new(A::Expr::IntLit {
133                value: 2,
134                base: A::IntBase::Dec,
135            }),
136        };
137        let n = normalize_expr(&e);
138        match n {
139            A::Expr::BinOp { op, .. } => assert_eq!(op, "+"),
140            _ => panic!("unexpected"),
141        }
142    }
143
144    #[test]
145    /// コレクション内部でも再帰的に正規化されることを確認する。
146    fn normalize_recurse_into_collections() {
147        // [(2 ^ -1)] のような入れ子でも再帰的に正規化されることを確認する
148        let e = A::Expr::ListLit {
149            items: vec![A::Expr::BinOp {
150                op: "^".into(),
151                left: Box::new(A::Expr::IntLit {
152                    value: 2,
153                    base: A::IntBase::Dec,
154                }),
155                right: Box::new(A::Expr::BinOp {
156                    op: "-".into(),
157                    left: Box::new(A::Expr::IntLit {
158                        value: 0,
159                        base: A::IntBase::Dec,
160                    }),
161                    right: Box::new(A::Expr::IntLit {
162                        value: 1,
163                        base: A::IntBase::Dec,
164                    }),
165                }),
166            }],
167        };
168        let n = normalize_expr(&e);
169        if let A::Expr::ListLit { items } = n {
170            if let A::Expr::BinOp { op, .. } = &items[0] {
171                assert_eq!(op, "**");
172                return;
173            }
174        }
175        panic!("not normalized in collection");
176    }
