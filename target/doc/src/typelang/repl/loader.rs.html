8use crate::evaluator::{eval_expr, Value};

13/// REPL/テスト共用: プログラムを型/値環境にロードする。
14///
15/// # 仕様
16/// - 各定義を正規化（`^` の負指数を `**` へ）し、型推論→既定化→評価の順で取り込みます。
17/// - 型注釈が与えられている場合は単一化で検証します。
18///
19/// # Examples
20/// ```
21/// use typelang::{parser, infer};
22/// let src = "square :: Num a => a -> a;\nlet square x = x * x;";
23/// let prog = parser::parse_program(src).unwrap();
24/// let mut tenv = infer::initial_env();
25/// let cenv = infer::initial_class_env();
26/// let mut venv = typelang::evaluator::initial_env();
27/// let loaded = typelang::repl::load_program_into_env(&prog, &mut tenv, &cenv, &mut venv).unwrap();
28/// assert_eq!(loaded, vec!["square"]);
29/// ```
30pub fn load_program_into_env(
31    prog: &A::Program,
32    type_env: &mut crate::typesys::TypeEnv,
33    class_env: &crate::typesys::ClassEnv,
34    value_env: &mut crate::evaluator::Env,
35) -> Result<Vec<String>, String> {
36    let mut type_env_tmp = type_env.clone_env();
37    let mut value_env_tmp = value_env.clone();
38    let mut loaded: Vec<String> = Vec::new();
39    for decl in &prog.decls {
40        let orig = if decl.params.is_empty() {
41            decl.expr.clone()
42        } else {
43            A::Expr::Lambda {
44                params: decl.params.clone(),
45                body: Box::new(decl.expr.clone()),
46            }
47        };
48        let body = normalize_expr(&orig);
49        let mut st = InferState {
50            supply: TVarSupply::new(),
51            subst: Default::default(),
52        };
53        let (s, q_rhs0) = match infer_expr(&type_env_tmp, class_env, &mut st, &body) {
54            Ok(ok) => ok,
55            Err(_e) => {
56                // 推論失敗時は値を評価し、代表型で一般化
57                let val = eval_expr(&body, &mut value_env_tmp).map_err(|e2| e2.to_string())?;
58                let sch = generalize(
59                    &type_env_tmp,
60                    crate::typesys::qualify(
61                        crate::typesys::Type::TCon(crate::typesys::TCon {
62                            name: match val {
63                                Value::Int(_) => "Int".into(),
64                                Value::Double(_) => "Double".into(),
65                                Value::Bool(_) => "Bool".into(),
66                                Value::Char(_) => "Char".into(),
67                                Value::String(_) => "[Char]".into(),
68                                _ => "()".into(),
69                            },
70                        }),
71                        vec![],
72                    ),
73                );
74                type_env_tmp.extend(decl.name.clone(), sch);
75                value_env_tmp.insert(decl.name.clone(), val);
76                loaded.push(decl.name.clone());
77                continue;
78            }
79        };
80        let q_rhs1 = crate::typesys::apply_subst_q(&s, &q_rhs0);
81        let is_fun = matches!(q_rhs1.r#type, crate::typesys::Type::TFun(_));
82        let q_rhs = if decl.signature.is_none() && !is_fun {
83            crate::typesys::apply_defaulting_simple(&q_rhs1)
84        } else {
85            q_rhs1
86        };
87        if let Some(sig) = &decl.signature {
88            let ty_anno = type_from_texpr(&sig.r#type);
89            let _s2 = unify(q_rhs.r#type.clone(), ty_anno)
90                .map_err(|e| format!("[{}] {}", e.code, e.message))?;
91        }
92        let sch = generalize(&type_env_tmp, q_rhs.clone());
93        type_env_tmp.extend(decl.name.clone(), sch);
94        let val = eval_expr(&body, &mut value_env_tmp).map_err(|e| e.to_string())?;
95        value_env_tmp.insert(decl.name.clone(), val);
96        loaded.push(decl.name.clone());
97    }
98    *type_env = type_env_tmp;
99    *value_env = value_env_tmp;
100    Ok(loaded)
101}
