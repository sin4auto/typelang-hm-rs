5/// ヘルプメッセージを表示する。
6pub(crate) fn print_help() {
7    println!("利用可能なコマンド:");
8    println!("  :help              ヘルプ（本メッセージ）");
9    println!("  :t EXPR            型を表示");
10    println!("  :type EXPR         :t と同じ");
11    println!("  :let DEF[; ...]    その場で定義（複数は ; 区切り）");
12    println!("  :load PATH         ファイルからロード");
13    println!("  :reload            直近ロードしたファイルを再読み込み");
14    println!("  :browse [PFX]      定義一覧（接頭辞フィルタ）");
15    println!("  :unset NAME        定義を削除");
16    println!("  :set default on|off 型表示の defaulting を切替");
17    println!("  :quit              終了");
18    println!();
19    println!("例:");
20    println!("  > :let square x = x * x");
21    println!("  > square 12             -- 144 を表示");
22    println!(r"  > :t \\x -> x ** 2      -- Fractional a => a -> a");
23    println!("  > 1 + 2                 -- 3 を表示、直近結果は it で参照可");
24    println!("  > it * 10               -- 30");
25}

27/// 値を REPL 用に表示する。
28pub(crate) fn print_value(v: &Value) {
29    match v {
30        Value::String(s) => println!("\"{}\"", s),
31        Value::Int(i) => println!("{}", i),
32        Value::Double(d) => println!("{}", d),
33        Value::Bool(b) => println!("{}", if *b { "True" } else { "False" }),
34        Value::Char(c) => println!("'{}'", c),
35        other => println!("{:?}", other),
36    }
37}
