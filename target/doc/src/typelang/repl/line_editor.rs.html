10/// 行入力の結果種別。
11pub enum ReadResult {
12    Line(String),
13    Eof,
14    Interrupted,
15}

22/// LineEditor の操作メソッド。
23impl LineEditor {
24    /// 履歴を読み込んで新しいエディタを生成する。
25    pub fn new() -> Self {
26        Self {

28        }

31    /// プロンプト付きで1行入力を読み取る。
32    pub fn read_line(&mut self, prompt: &str) -> io::Result<ReadResult> {

45        self.history.add(entry);
46    }

49    pub fn save_history(&self) -> io::Result<()> {
50        self.history.save()
51    }

61#[cfg(not(unix))]
62impl LineEditor {
63    /// UNIX以外の環境向けに行入力を読む。
64    fn read_line_fallback(&mut self, prompt: &str) -> io::Result<ReadResult> {
65        let mut stdout = io::stdout();
66        write!(stdout, "{}", prompt)?;
67        stdout.flush()?;
68        let mut line = String::new();
69        let bytes = io::stdin().read_line(&mut line)?;
70        if bytes == 0 {
71            return Ok(ReadResult::Eof);
72        }
73        if line.ends_with('\n') {
74            line.pop();
75        }
76        if line.ends_with('\r') {
77            line.pop();
78        }
79        Ok(ReadResult::Line(line))
80    }
81}

83#[cfg(unix)]
84impl LineEditor {
85    /// UNIX環境で端末をRawモードにして読み取る。
86    fn read_line_unix(&mut self, prompt: &str) -> io::Result<ReadResult> {
87        let _raw = RawMode::new()?;
88        let mut stdout = io::stdout();
89        write!(stdout, "{}", prompt)?;
90        stdout.flush()?;
91
92        let mut buffer: Vec<char> = Vec::new();
93        let mut cursor: usize = 0;
94        let mut history_index = self.history.len();
95        let mut saved_current: Option<Vec<char>> = None;
96
97        let stdin = io::stdin();
98        let mut stdin = stdin.lock();
99        loop {
100            let mut byte = [0u8; 1];
101            if stdin.read(&mut byte)? == 0 {
102                return Ok(ReadResult::Eof);
103            }
104            let b = byte[0];
105            match b {
106                b'\n' | b'\r' => {
107                    write!(stdout, "\r\n")?;
108                    stdout.flush()?;
109                    let line: String = buffer.into_iter().collect();
110                    return Ok(ReadResult::Line(line));
111                }
112                0x03 => {
113                    write!(stdout, "^C\r\n")?;
114                    stdout.flush()?;
115                    return Ok(ReadResult::Interrupted);
116                }
117                0x04 => {
118                    if buffer.is_empty() {
119                        return Ok(ReadResult::Eof);
120                    }
121                }
122                0x7f | 0x08 => {
123                    if cursor > 0 {
124                        cursor -= 1;
125                        buffer.remove(cursor);
126                        history_index = self.history.len();
127                        saved_current = None;
128                        refresh_line(&mut stdout, prompt, &buffer, cursor)?;
129                    }
130                }
131                0x1b => {
132                    let context = EscapeContext {
133                        history: &self.history,
134                        history_index: &mut history_index,
135                        saved_current: &mut saved_current,
136                    };
137                    handle_escape(
138                        &mut stdin,
139                        &mut stdout,
140                        prompt,
141                        &mut buffer,
142                        &mut cursor,
143                        context,
144                    )?;
145                }
146                _ => {
147                    if let Some(ch) = read_utf8_char(b, &mut stdin)? {
148                        if ch.is_control() {
149                            continue;
150                        }
151                        buffer.insert(cursor, ch);
152                        cursor += 1;
153                        history_index = self.history.len();
154                        saved_current = None;
155                        refresh_line(&mut stdout, prompt, &buffer, cursor)?;
156                    }
157                }
158            }
159        }
160    }
161}

163/// UTF-8の先頭バイト列から文字を復元する。
164fn read_utf8_char<R: Read>(first: u8, reader: &mut R) -> io::Result<Option<char>> {
165    let width = match first {
166        0x00..=0x7f => 1,
167        0xc2..=0xdf => 2,
168        0xe0..=0xef => 3,
169        0xf0..=0xf4 => 4,
170        _ => return Ok(None),
171    };
172    let mut buf = [0u8; 4];
173    buf[0] = first;
174    for idx in 1..width {
175        reader.read_exact(&mut buf[idx..idx + 1])?;
176    }
177    match std::str::from_utf8(&buf[..width]) {
178        Ok(s) => Ok(s.chars().next()),
179        Err(_) => Ok(None),
180    }
181}

184/// エスケープシーケンス処理に必要な状態をまとめる。
185struct EscapeContext<'a> {
186    history: &'a History,
187    history_index: &'a mut usize,
188    saved_current: &'a mut Option<Vec<char>>,
189}
190
191#[cfg(unix)]
192/// 端末から届いたエスケープシーケンスを処理する。
193fn handle_escape<R: Read, W: Write>(
194    reader: &mut R,
195    writer: &mut W,
196    prompt: &str,
197    buffer: &mut Vec<char>,
198    cursor: &mut usize,
199    context: EscapeContext<'_>,
200) -> io::Result<()> {
201    let EscapeContext {
202        history,
203        history_index,
204        saved_current,
205    } = context;
206    let mut seq = [0u8; 2];
207    if reader.read_exact(&mut seq[..1]).is_err() {
208        return Ok(());
209    }
210    if seq[0] != b'[' {
211        return Ok(());
212    }
213    if reader.read_exact(&mut seq[1..2]).is_err() {
214        return Ok(());
215    }
216    match seq[1] {
217        b'A' => {
218            if *history_index > 0 {
219                if *history_index == history.len() {
220                    *saved_current = Some(buffer.clone());
221                }
222                *history_index -= 1;
223                if let Some(entry) = history.get(*history_index) {
224                    *buffer = entry.chars().collect();
225                    *cursor = buffer.len();
226                    refresh_line(writer, prompt, buffer, *cursor)?;
227                }
228            }
229        }
230        b'B' => {
231            if *history_index < history.len() {
232                *history_index += 1;
233                let restored = if *history_index == history.len() {
234                    saved_current.clone().unwrap_or_default()
235                } else {
236                    history.get(*history_index).unwrap().chars().collect()
237                };
238                *buffer = restored;
239                *cursor = buffer.len();
240                refresh_line(writer, prompt, buffer, *cursor)?;
241            }
242        }
243        b'C' => {
244            if *cursor < buffer.len() {
245                *cursor += 1;
246                write!(writer, "\x1b[C")?;
247                writer.flush()?;
248            }
249        }
250        b'D' => {
251            if *cursor > 0 {
252                *cursor -= 1;
253                write!(writer, "\x1b[D")?;
254                writer.flush()?;
255            }
256        }
257        _ => {}
258    }
259    Ok(())
260}

262#[cfg(unix)]
263/// 1行分の表示を再描画する。
264fn refresh_line<W: Write>(
265    writer: &mut W,
266    prompt: &str,
267    buffer: &[char],
268    cursor: usize,
269) -> io::Result<()> {
270    let rendered: String = buffer.iter().collect();
271    write!(writer, "\r{}{}", prompt, rendered)?;
272    write!(writer, "\x1b[K")?;
273    let total = prompt.chars().count() + buffer.len();
274    let target = prompt.chars().count() + cursor;
275    if total > target {
276        write!(writer, "\x1b[{}D", total - target)?;
277    }
278    writer.flush()
279}

281/// 入力履歴を保持・永続化する構造体。
282struct History {
283    entries: Vec<String>,
284    path: Option<PathBuf>,
285    max_entries: usize,
286}
287
288impl History {
289    /// 保存済みの履歴を読み込む。
290    fn load() -> Self {
291        let path = history_path();
292        let entries = path
293            .as_ref()
294            .and_then(|p| fs::read_to_string(p).ok())
295            .map(|content| content.lines().map(|s| s.to_string()).collect())
296            .unwrap_or_default();
297        Self {
298            entries,
299            path,
300            max_entries: 1000,
301        }
302    }
303
304    /// 履歴に項目を追加する。
305    fn add(&mut self, entry: &str) {
306        let trimmed = entry.trim();
307        if trimmed.is_empty() {
308            return;
309        }
310        if self.entries.last().map(|s| s.as_str()) == Some(trimmed) {
311            return;
312        }
313        if self.entries.len() == self.max_entries {
314            self.entries.remove(0);
315        }
316        self.entries.push(trimmed.to_string());
317    }
318
319    /// 登録済みの履歴数を返す。
320    fn len(&self) -> usize {
321        self.entries.len()
322    }
323
324    /// 履歴の特定位置を参照する。
325    fn get(&self, idx: usize) -> Option<&str> {
326        self.entries.get(idx).map(|s| s.as_str())
327    }
328
329    /// 履歴をファイルに保存する。
330    fn save(&self) -> io::Result<()> {
331        let Some(path) = &self.path else {
332            return Ok(());
333        };
334        if let Some(parent) = path.parent() {
335            fs::create_dir_all(parent)?;
336        }
337        let mut file = fs::File::create(path)?;
338        for entry in &self.entries {
339            writeln!(file, "{}", entry)?;
340        }
341        Ok(())
342    }
343}

345/// 履歴ファイルの探索および環境変数の優先順位を決める。
346fn history_path() -> Option<PathBuf> {
347    if let Some(path) = env::var_os("TYPELANG_HISTORY_FILE") {
348        return Some(PathBuf::from(path));
349    }
350    env::var_os("HOME")
351        .or_else(|| env::var_os("USERPROFILE"))
352        .map(PathBuf::from)
353        .map(|home| home.join(".typelang_repl_history"))
354}

356#[cfg(unix)]
357/// 端末設定をRawモードへ切り替えるためのガード。
358struct RawMode {
359    original: Termios,
360}
361
362#[cfg(unix)]
363impl RawMode {
364    /// 標準入力をRawモードへ切り替える。
365    fn new() -> io::Result<Self> {
366        let fd = 0; // stdin
367        let mut termios = Termios::default();
368        if unsafe { tcgetattr(fd, &mut termios as *mut _) } != 0 {
369            return Err(io::Error::last_os_error());
370        }
371        let mut raw = termios;
372        // `cfmakeraw` に任せることで OS ごとの差分を吸収して Raw モードへ移行する。
373        unsafe {
374            cfmakeraw(&mut raw as *mut _);
375        }
376        if unsafe { tcsetattr(fd, TCSANOW, &raw as *const _) } != 0 {
377            return Err(io::Error::last_os_error());
378        }
379        Ok(Self { original: termios })
380    }
381}
382
383#[cfg(unix)]
384impl Drop for RawMode {
385    /// スコープ終了時に元の端末設定へ戻す。
386    fn drop(&mut self) {
387        let fd = 0;
388        unsafe {
389            let _ = tcsetattr(fd, TCSANOW, &self.original as *const _);
390        }
391    }
392}
